### この章は、省略なしの英語で書かれています。

# スキップを禁止して、ロゴ表示をスキップできなくする
@skip disable

# ステージを白い背景でクリアします
@bg file=white.png duration=0

# ロゴを表示します
@bg file=logo.png 0.5

# 2秒待ちます
@wait duration=2.0

# フォントライセンスを表示します
@bg file=font.png duration=0.5

# 2秒待ちます
@wait duration=2.0

# 暗転します
@bg black.png 0.5

# スキップを許可します
@skip enable

###
### メニュー
###

# BGMを再生する
@音量 トラック=音楽 音量=1.0
@音楽 ファイル=01.ogg

# メニューの背景を表示する
@背景 ファイル=title.png 1.0

# メニューを表示する
:MENU
@メニュー ファイル=title.txt

# メニューで"START"が選択されたとき、ここにジャンプしてくる
:START

# シーン1にジャンプする
@ジャンプ 行き先=シーン1

# メニューで"LOAD"が選択されたとき、ここにジャンプしてくる
:LOAD
@メニュー ファイル=load.txt cancel
@ジャンプ 行き先=MENU

# メニューで"CONFIG"が選択されたとき、ここにジャンプしてくる
:CONFIG
@メニュー ファイル=system.txt cancel
@ジャンプ 行き先=MENU





:シーン1

#BGMをストップする
@vol bgm 0 1.0
@wait 1.0

@bgm stop

#背景を真っ暗にする
@bg #000000 1.0

#背景を変更する
@bg coast.png 1.0

# メッセージボックスをフェードインさせる
@anime msgbox-show.txt forcemsgbox

# ページ
\===
背景を変更しますか？
\---

@ichoose L1 はい L2 いいえ

:L1

# メッセージボックスをフェードアウトさせる
@anime msgbox-out.txt forcemsgbox

# 背景を変更する
@bg roof.png 1.0

# メッセージボックスの文字を消去する
# （フェード中に前の文字が映るのを防ぐため）
\erase

# メッセージボックスをフェードインさせる
# （フェードアウト後、フェードインを忘れると文字が表示されなくなる）
@anime msgbox-show.txt forcemsgbox

:L2

# ページ
\===
キャラクタを表示します。
\---

@ch center 001-happy.png 1.0
