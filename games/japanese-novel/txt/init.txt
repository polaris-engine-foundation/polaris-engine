# メニューの背景を表示する
@背景 ファイル=title.png 0

# メニューを表示する
:MENU
@メニュー ファイル=title.txt

# メニューで"START"が選択されたとき、ここにジャンプしてくる
:START

# シーン1にジャンプする
@ジャンプ 行き先=シーン1

# メニューで"LOAD"が選択されたとき、ここにジャンプしてくる
:LOAD
@メニュー ファイル=load.txt cancel
@ジャンプ 行き先=MENU

# メニューで"CONFIG"が選択されたとき、ここにジャンプしてくる
:CONFIG
@メニュー ファイル=system.txt cancel
@ジャンプ 行き先=MENU


# メニューで"FONT"が選択されたとき、ここにジャンプしてくる
:FONT

@click
